            float sum_pow = 0.0;

            scanf("%d", &lower);
            printf("Enter upper range: ");
            scanf("%d", &upper);
        
            printf("Armstrong numbers between %d and %d are: ", lower, upper);
        
            for (i = lower + 1; i < upper; ++i)
            {
                temp1 = i;
                temp2 = i;
        
             
                while (temp1 != 0)
                {
                    temp1 /= 10;
                    ++num;
                }
        
              
                while (temp2 != 0) {
                    rem = temp2 % 10;
                    sum_pow += pow(rem, num);
                    temp2 /= 10;
                }
        
               
                if ((int)sum_pow == i) {
                    printf("%d ", i);
                }
        
                num = 0;
                sum_pow = 0;
            }
            return 0;
        }
